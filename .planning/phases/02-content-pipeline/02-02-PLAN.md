  <name>Task 1: Implement pipeline/backlog.py — all DB operations</name>
  <files>pipeline/backlog.py</files>
  <behavior>
    Module-level constant: PROBATION_THRESHOLD = 25

    init_backlog_tables(conn: sqlite3.Connection) -> None:
    - Executes the DDL for backlog_stories, backlog_tweets, niche_state using conn.executescript()
    - This is a duplicate of the DDL in analysis/db.py — backlog.py owns the canonical DDL; analysis/db.py calls this function
    - Tables use CREATE TABLE IF NOT EXISTS

    insert_story(conn, item: dict) -> bool:
    - INSERT OR IGNORE INTO backlog_stories with status='pending', scraped_at=utcnow ISO
    - item keys: id, channel, subreddit, title, body, score, word_count
    - Returns True if inserted (rowcount > 0), False if duplicate

    insert_tweet(conn, item: dict) -> bool:
    - INSERT OR IGNORE INTO backlog_tweets with status='pending', scraped_at=utcnow ISO
    - item keys: tweet_id, channel, username, tweet_text, likes, retweets
    - Returns True if inserted, False if duplicate

    approve_item(conn, table: str, item_id: str, channel: str) -> None:
    - UPDATE SET status='approved', approved_at=utcnow WHERE id/tweet_id=item_id
    - Also calls increment_reviewed_count(conn, channel)
    - table param is either 'backlog_stories' or 'backlog_tweets'
    - Use the correct primary key: 'id' for stories, 'tweet_id' for tweets
    - Private helper _pk(table) -> str returns the primary key column name

    reject_item(conn, table: str, item_id: str, channel: str) -> None:
    - UPDATE SET status='rejected' WHERE id/tweet_id=item_id
    - Also calls increment_reviewed_count(conn, channel) — rejection counts as a review

    mark_used(conn, table: str, item_id: str) -> None:
    - UPDATE SET status='used', used_at=utcnow WHERE id/tweet_id=item_id

    get_approved_stories(conn, channel: str) -> list[sqlite3.Row]:
    - SELECT * FROM backlog_stories WHERE channel=? AND status='approved' ORDER BY approved_at ASC

    get_approved_tweets(conn, channel: str) -> list[sqlite3.Row]:
    - SELECT * FROM backlog_tweets WHERE channel=? AND status='approved' ORDER BY approved_at ASC

    get_pending_stories(conn, channel: str) -> list[sqlite3.Row]:
    - SELECT * FROM backlog_stories WHERE channel=? AND status='pending' ORDER BY scraped_at ASC

    get_pending_tweets(conn, channel: str) -> list[sqlite3.Row]:
    - SELECT * FROM backlog_tweets WHERE channel=? AND status='pending' ORDER BY scraped_at ASC

    get_status_counts(conn, channel: str) -> dict[str, dict[str, int]]:
    - Returns {channel: {pending: N, approved: N, rejected: N, used: N}} for given channel
    - If channel == 'all', returns counts for all channels merged by channel slug

    maybe_auto_approve(conn, table: str, item_id: str, channel: str) -> bool:
    - Calls _ensure_niche_state(conn, channel) first (INSERT OR IGNORE)
    - Checks manually_reviewed_count >= PROBATION_THRESHOLD
    - If yes: calls approve_item() but does NOT increment reviewed count again (pass skip_increment=True)
    - Returns True if auto-approved, False if still in probation
    - NOTE: maybe_auto_approve does NOT call increment_reviewed_count — items auto-approved without
      manual review don't advance probation count

    Wait — revise: maybe_auto_approve should NOT call approve_item (which increments count). Instead
    directly UPDATE SET status='approved', approved_at=utcnow for auto-approved items.

    increment_reviewed_count(conn, channel: str) -> None:
    - _ensure_niche_state(conn, channel) first
    - UPDATE niche_state SET manually_reviewed_count = manually_reviewed_count + 1 WHERE channel=?

    get_probation_remaining(conn, channel: str) -> int:
    - _ensure_niche_state(conn, channel) first
    - Returns max(0, PROBATION_THRESHOLD - manually_reviewed_count)

    _ensure_niche_state(conn, channel: str) -> None (private):
    - INSERT OR IGNORE INTO niche_state (channel, manually_reviewed_count) VALUES (?, 0)

    _pk(table: str) -> str (private):
    - Returns 'id' for 'backlog_stories', 'tweet_id' for 'backlog_tweets'
    - Raises ValueError for unknown table

    All functions use type hints. Module-level logger = logging.getLogger(__name__).
    datetime.utcnow().isoformat() for all timestamps.
    Use conn.commit() only in insert functions; let callers manage transactions for bulk ops.
